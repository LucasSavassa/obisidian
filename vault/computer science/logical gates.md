logical gates are electronic components that implement [[boolean logic]]
# not
if ( a = 1 )
	return 0
else
	return 1

# and
if ( a = 1 and b = 1 )
	return 1
else
	return 0	
![[gate and]]
# or
if ( a = 1 or b = 1)
	return 1
else
	return 0
![[gate or]]
# nand
if ( a = 1 and b = 1 )
    return 0
else
	return 1

![[gate nand]]

# xor
if (
	( a = 1 and b = 0 ) or
	( a = 0 and b = 1 )
)
	return 1
else
	return 0
![[gate xor]]
